-- Editorial
create or alter procedure usp_listarEditorial
AS
SELECT * FROM tb_editorial;

CREATE PROCEDURE usp_insertarEditorial
    @CodigoEditorial INT,
    @Nombre VARCHAR(150),
	@Pais VARCHAR(150)
AS
INSERT INTO tb_editorial VALUES (@CodigoEditorial, @Nombre, @Pais);
go

CREATE PROCEDURE usp_actualizarEditorial
    @CodigoEditorial INT,
    @Nombre VARCHAR(150),
	@Pais VARCHAR(150)
AS
UPDATE tb_editorial
SET Nombre = @Nombre,
    Pais = @Pais
WHERE CodigoEditorial = @CodigoEditorial;

---Venta de libros
CREATE TABLE tb_venta (
    IdVenta INT IDENTITY(1,1) PRIMARY KEY,
    CodigoLibro VARCHAR(7) NOT NULL,
    Cantidad INT NOT NULL,
    PrecioUnitario DECIMAL(10,2) NOT NULL,
    Total DECIMAL(10,2) NOT NULL,
    FechaVenta DATETIME NOT NULL DEFAULT GETDATE(),

    FOREIGN KEY (CodigoLibro) REFERENCES tb_libro(CodigoLibro)
);

create or alter procedure listadoVenta 
as 
begin
select * from tb_venta
end






CREATE PROCEDURE usp_registrarVenta
    @CodigoLibro VARCHAR(7),
    @Cantidad INT
AS
BEGIN
    BEGIN TRANSACTION;

    BEGIN TRY
        DECLARE @StockActual INT;
        DECLARE @PrecioUnitario DECIMAL(10,2);

        -- Verificar si existe el libro
        IF NOT EXISTS (SELECT 1 FROM tb_libro WHERE CodigoLibro = @CodigoLibro)
        BEGIN
            THROW 50001, 'El libro no existe.', 1;
        END

        -- Obtener stock actual y precio
        SELECT @StockActual = Stock, @PrecioUnitario = Precio
        FROM tb_libro
        WHERE CodigoLibro = @CodigoLibro;

        -- Validar stock
        IF @StockActual < @Cantidad
        BEGIN
            THROW 50002, 'Stock insuficiente para completar la venta.', 1;
        END

        -- Actualizar stock
        UPDATE tb_libro
        SET Stock = Stock - @Cantidad
        WHERE CodigoLibro = @CodigoLibro;

        -- Registrar venta
        INSERT INTO tb_venta (CodigoLibro, Cantidad, PrecioUnitario, Total)
        VALUES (@CodigoLibro, @Cantidad, @PrecioUnitario, @Cantidad * @PrecioUnitario);

        COMMIT;
    END TRY
    BEGIN CATCH
        ROLLBACK;
        THROW;
    END CATCH
END;


exec usp_registrarVenta 'LIB002', 2

CREATE OR ALTER PROCEDURE usp_listarLibroPaginado
    @search VARCHAR(100) = '',
    @page INT = 1,
    @pageSize INT = 5
AS
BEGIN
    SELECT l.CodigoLibro, l.Titulo, l.ISBN, e.Nombre, l.Precio, l.Stock
    FROM tb_libro l
    INNER JOIN tb_editorial e ON l.CodigoEditorial = e.CodigoEditorial
    WHERE l.Titulo LIKE '%' + @search + '%' OR l.CodigoLibro LIKE '%' + @search + '%'
    ORDER BY l.CodigoLibro
    OFFSET (@page - 1) * @pageSize ROWS
    FETCH NEXT @pageSize ROWS ONLY;
END;
GO

CREATE OR ALTER PROCEDURE usp_contarLibros
    @search VARCHAR(100) = ''
AS
BEGIN
    SELECT COUNT(*) 
    FROM tb_libro 
    WHERE Titulo LIKE '%' + @search + '%' OR CodigoLibro LIKE '%' + @search + '%';
END;
GO
